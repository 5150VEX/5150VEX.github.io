113 ProfileSetpoint<Units> calculate(const au::Quantity<au::Seconds, double> elapsed_time) const {
114 // integration constant for the coasting segment
115 au::Quantity<Units, double> const_1 =
116 -m_max_velocity * accel_cutoff
117 + (m_max_acceleration / 2) * au::int_pow<2>(accel_cutoff);
118
119 // integration constant for the deceleration segment
120 au::Quantity<Units, double> const_2 =
121 -m_max_acceleration * total_time * coast_cutoff
122 + (m_max_acceleration / 2) * au::int_pow<2>(coast_cutoff)
123 + m_max_velocity * coast_cutoff + const_1;
124
125 ProfileSetpoint<Units> setpoint = ProfileSetpoint<Units>(au::ZERO, au::ZERO, au::ZERO);
126
127 switch (this->stage(elapsed_time)) {
128 case TrapezoidProfileStage::Accelerating:
129 setpoint = ProfileSetpoint(
130 (m_max_acceleration / 2) * au::int_pow<2>(elapsed_time),
131 m_max_acceleration * elapsed_time,
132 m_max_acceleration
133 );
134 break;
135 case TrapezoidProfileStage::Coasting:
136 setpoint = ProfileSetpoint(
137 m_max_velocity * elapsed_time + const_1,
138 m_max_velocity,
139 m_max_acceleration - m_max_acceleration // long ass way to say 0 bc complier complains
140 );
141 break;
142 case TrapezoidProfileStage::Decelerating:
143 setpoint = ProfileSetpoint(
144 m_max_acceleration * total_time * elapsed_time
145 - (m_max_acceleration / 2) * au::int_pow<2>(elapsed_time)
146 + const_2,
147 m_max_acceleration * (total_time - elapsed_time),
148 -m_max_acceleration
149 );
150 break;
151 case TrapezoidProfileStage::Done:
152 setpoint = ProfileSetpoint<Units>(m_total_distance, au::ZERO, au::ZERO);
153 break;
154 default:
155 // to stop the compiler from complaining
156 setpoint = ProfileSetpoint<Units>(m_total_distance, au::ZERO, au::ZERO);
157 }
158
159 if (invert) {
